[CanExamineOrWhatever] = $SET_HOOK_PARAM( 5, "APPLY" ); $SET_HOOK_PARAM( 6, 0 ); $RETURN "Y";
[ExamineOrWhatever] = $VAR item; $VAR advant; $VAR adj;
-item = "Advantage: Defensive Strike"; advant = $SA_CHARACTER_GET("advantageDefensiveStrike");
-$IF (advant == "-?-?-") {advant = "0";}; adj = $SA_PARAM_GET(); advant = advant +# adj;
-$TAKE_CHAR_ITEM($CharacterContext(),item);
-$SET_CHARACTER_SA($CharacterContext(),"advantageDefensiveStrike",advant);
\(END)

\(BEGIN)
name = advantageDefensiveStrike
{CharDisplayStatus] = $VAR prm; $VAR p5;
-prm = $SA_PARAM_GET(); p5 = $GET_HOOK_PARAM(5);
-$IF (p5 == "DiSPLAY") {$SET_HOOK_PARAM(7,"Advantage: Defensive Strike +"+prm); $RETURN "1";};
\(END)
